019package org.tynamo.security.components;
020
021import java.io.IOException;
022import java.net.URL;
023
024import org.apache.shiro.authc.AuthenticationException;
025import org.apache.shiro.authc.IncorrectCredentialsException;
026import org.apache.shiro.authc.LockedAccountException;
027import org.apache.shiro.authc.UnknownAccountException;
028import org.apache.shiro.authc.UsernamePasswordToken;
029import org.apache.shiro.subject.Subject;
030import org.apache.shiro.util.StringUtils;
031import org.apache.tapestry5.BindingConstants;
032import org.apache.tapestry5.ComponentResources;
033import org.apache.tapestry5.EventConstants;
034import org.apache.tapestry5.ValidationException;
035import org.apache.tapestry5.annotations.OnEvent;
036import org.apache.tapestry5.annotations.Parameter;
037import org.apache.tapestry5.annotations.Property;
038import org.apache.tapestry5.ioc.Messages;
039import org.apache.tapestry5.ioc.annotations.Inject;
040import org.apache.tapestry5.ioc.annotations.Symbol;
041import org.apache.tapestry5.services.Cookies;
042import org.apache.tapestry5.services.PageRenderLinkSource;
043import org.apache.tapestry5.services.RequestGlobals;
044import org.apache.tapestry5.services.Response;
045import org.slf4j.Logger;
046import org.slf4j.LoggerFactory;
047import org.tynamo.security.SecuritySymbols;
048import org.tynamo.security.internal.services.LoginContextService;
049import org.tynamo.security.services.SecurityService;
050
051/**
052 * Login form component
053 *
054 */
055public class LoginForm
056{
057
058        private static final Logger logger = LoggerFactory.getLogger(LoginForm.class);
059
060        private static final String LOGIN_FORM_ID = "tynamoLoginForm";
061
062        @Property
063        private String tynamoLogin;
064
065        @Property
066        private String tynamoPassword;
067
068        @Property
069        private boolean tynamoRememberMe;
070
071        private String loginMessage;
072
073        /** Use ^ for current page the component is contained in, empty for configured success url */
074        @Parameter(defaultPrefix = BindingConstants.LITERAL)
075        private String successURL;
076
077        @Inject
078        private Messages messages;
079
080        @Inject
081        private Response response;
082
083        @Inject
084        private RequestGlobals requestGlobals;
085
086        @Inject
087        private SecurityService securityService;
088
089        @Inject
090        private LoginContextService loginContextService;
091
092        @Inject
093        private PageRenderLinkSource pageRenderLinkSource;
094
095        @Inject
096        private ComponentResources componentResources;
097
098        @Inject
099        private Cookies cookies;
100
101        @Inject
102        @Symbol(SecuritySymbols.REDIRECT_TO_SAVED_URL)
103        private boolean redirectToSavedUrl;
104
105        @OnEvent(value = EventConstants.VALIDATE, component = LOGIN_FORM_ID)
106        public void attemptToLogin() throws ValidationException
107        {
108
109                Subject currentUser = securityService.getSubject();
110
111                if (currentUser == null)
112                {
113                        throw new IllegalStateException("Subject can`t be null");
114                }
115
116                UsernamePasswordToken token = new UsernamePasswordToken(tynamoLogin, tynamoPassword);
117                token.setRememberMe(tynamoRememberMe);
118
119
120                try
121                {
122                        currentUser.login(token);
123                } catch (UnknownAccountException e)
124                {
125                        loginMessage = messages.get("AccountDoesNotExists");
126                } catch (IncorrectCredentialsException e)
127                {
128                        loginMessage = messages.get("WrongPassword");
129                } catch (LockedAccountException e)
130                {
131                        loginMessage = messages.get("AccountLocked");
132                } catch (AuthenticationException e)
133                {
134                        loginMessage = messages.get("AuthenticationError");
135                }
136
137                if (loginMessage != null)
138                {
139                    throw new ValidationException(loginMessage);
140                }
141        }
142
143        @OnEvent(value = EventConstants.SUCCESS, component = LOGIN_FORM_ID)
144        public Object onSuccessfulLogin() throws IOException
145        {
146                if (StringUtils.hasText(successURL)) {
147                        if ("^".equals(successURL))
148                                return pageRenderLinkSource.createPageRenderLink(componentResources.getPage().getClass());
149                        return new URL(successURL);
150                }
151
152                if (redirectToSavedUrl) {
153                        String requestUri = loginContextService.getSuccessPage();
154                        if (!requestUri.startsWith("/") && !requestUri.startsWith("http")) {
155                            requestUri = "/" + requestUri;
156                        }
157                        loginContextService.redirectToSavedRequest(requestUri);
158                        return null;
159                }
160                return loginContextService.getSuccessPage();
161        }
162
163        public void setLoginMessage(String loginMessage)
164        {
165                this.loginMessage = loginMessage;
166        }
167
168        public String getLoginMessage()
169        {
170                if (StringUtils.hasText(loginMessage))
171                {
172                        return loginMessage;
173                } else
174                {
175                        return " ";
176                }
177        }
178
179}
