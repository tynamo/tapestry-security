19  /**
20   * Transform components based on annotation.
21   * <p/>
22   * Support annotation on method.
23   * <p/>
24   * The following rules
25   * <ul>
26   * <li>Annotations on methods are <b>not</b> inherited.</li>
27   * <li>The annotations only in target class, unlike services </li>
28   * <ul>
29   * <p/>
30   *
31   * @see org.tynamo.security.services.SecurityModule#buildSecurityFilter(org.slf4j.Logger,
32   *      org.apache.tapestry5.services.ComponentEventLinkEncoder,
33   *      org.apache.tapestry5.services.ComponentClassResolver,
34   *      org.tynamo.security.services.ClassInterceptorsCache)
35   */
36  public class ShiroAnnotationWorker implements ComponentClassTransformWorker2
37  {
38  	private final Environment environment;
39  
40  	public ShiroAnnotationWorker(Environment environment)
41  	{
42  		this.environment = environment;
43  	}
44  
45  	@Override
46  	public void transform(PlasticClass plasticClass, TransformationSupport support, MutableComponentModel model)
47  	{
48  		for (Class<? extends Annotation> annotationClass : AopHelper.getAutorizationAnnotationClasses())
49  		{
50  			List<PlasticMethod> methodsToTransform = plasticClass.getMethodsWithAnnotation(annotationClass);
51  
52  			for (PlasticMethod tm : methodsToTransform) processTransform(tm, tm.getAnnotation(annotationClass));
53  		}
54  	}
55  
56  	private void processTransform(PlasticMethod tm, Annotation annotation)
57  	{
58  		final SecurityInterceptor interceptor = new DefaultSecurityInterceptor(annotation);
59  
60  		MethodAdvice advice = new MethodAdvice()
61  		{
62  			public void advise(MethodInvocation invocation)
63  			{
64  				environment.push(MethodInvocation.class, invocation);
65  				try
66  				{
67  					interceptor.intercept();
68  				}
69  				finally
70  				{
71  					environment.pop(MethodInvocation.class);
72  				}
73  				invocation.proceed();
74  			}
75  		};
76  
77  		tm.addAdvice(advice);
78  
79  	}
80  
81  }
