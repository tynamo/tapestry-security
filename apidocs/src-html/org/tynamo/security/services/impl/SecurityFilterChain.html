001package org.tynamo.security.services.impl;
002
003import org.apache.shiro.util.AntPathMatcher;
004import org.apache.shiro.util.PatternMatcher;
005import org.apache.tapestry5.ioc.services.PipelineBuilder;
006import org.apache.tapestry5.services.HttpServletRequestFilter;
007import org.apache.tapestry5.services.HttpServletRequestHandler;
008import org.slf4j.Logger;
009import org.tynamo.security.shiro.AccessControlFilter;
010
011import javax.servlet.Filter;
012import java.util.ArrayList;
013import java.util.List;
014
015public class SecurityFilterChain {
016
017        /**
018         * Default PatternMatcher for backward compatibility
019         */
020        @Deprecated
021        private static final PatternMatcher defaultPatternMatcher = new AntPathMatcher();
022
023        private String path;
024        private HttpServletRequestHandler handler;
025        private PatternMatcher patternMatcher;
026
027        public SecurityFilterChain(String path, HttpServletRequestHandler handler, PatternMatcher patternMatcher) {
028                this.path = path;
029                this.handler = handler;
030                this.patternMatcher = patternMatcher;
031        }
032
033        /**
034         * @deprecated in 0.4.5 Use {@link #SecurityFilterChain(String, org.apache.tapestry5.services.HttpServletRequestHandler, org.apache.shiro.util.PatternMatcher)} instead
035         */
036        @Deprecated
037        public SecurityFilterChain(String path, HttpServletRequestHandler handler) {
038                this(path, handler, defaultPatternMatcher);
039        }
040
041        public HttpServletRequestHandler getHandler() {
042                return handler;
043        }
044
045        public boolean matches(String requestURI) {
046                return patternMatcher.matches(path, requestURI.toLowerCase());
047        }
048
049        public static class Builder {
050
051                private PipelineBuilder pipelineBuilder;
052                private String path;
053                private List<HttpServletRequestFilter> filters = new ArrayList<HttpServletRequestFilter>();
054                private Logger logger;
055                private  PatternMatcher patternMatcher;
056
057                public Builder(Logger logger, PipelineBuilder pipelineBuilder, String path, PatternMatcher patternMatcher) {
058                        this.logger = logger;
059                        this.pipelineBuilder = pipelineBuilder;
060                        this.path = path;
061                        this.patternMatcher = patternMatcher;
062                }
063
064                public Builder add(Class<HttpServletRequestFilter> filterType) {
065                        try {
066                                filters.add(filterType.newInstance());
067                        } catch (InstantiationException e) {
068                                throw new RuntimeException("Couldn't instantiate a filter while building a security chain for path '" + path + "': ", e);
069                        } catch (IllegalAccessException e) {
070                                throw new RuntimeException("Couldn't instantiate a filter while building a security chain for path '" + path + "': ", e);
071                        }
072                        return this;
073                }
074
075                public Builder add(Filter filter) {
076                        if (filter instanceof AccessControlFilter) add((AccessControlFilter) filter, null);
077                        else filters.add(new HttpServletRequestFilterWrapper(filter));
078                        return this;
079                }
080
081                public Builder add(AccessControlFilter filter, String config) {
082                        filter.addConfig(config);
083                        filters.add(new HttpServletRequestFilterWrapper(filter));
084                        return this;
085                }
086
087                public SecurityFilterChain build() {
088                        return new SecurityFilterChain(path, pipelineBuilder.build(logger, HttpServletRequestHandler.class, HttpServletRequestFilter.class, filters), patternMatcher);
089                }
090        }
091}
