19  package org.tynamo.security.shiro.authc;
20  
21  import javax.servlet.ServletRequest;
22  import javax.servlet.ServletResponse;
23  
24  import org.apache.shiro.subject.Subject;
25  import org.apache.shiro.web.util.WebUtils;
26  import org.tynamo.security.internal.services.LoginContextService;
27  import org.tynamo.security.shiro.AccessControlFilter;
28  
29  /**
30   * Base class for all Filters that require the current user to be authenticated. This class encapsulates the
31   * logic of checking whether a user is already authenticated in the system while subclasses are required to perform
32   * specific logic for unauthenticated requests.
33   *
34   * @since 0.9
35   */
36  public abstract class AuthenticationFilter extends AccessControlFilter {
37  	public AuthenticationFilter(LoginContextService loginContextService) {
38  		super(loginContextService);
39  	}
40  
41      /**
42       * Determines whether the current subject is authenticated.
43       * <p/>
44       * The default implementation {@link #getSubject(javax.servlet.ServletRequest, javax.servlet.ServletResponse) acquires}
45       * the currently executing Subject and then returns
46       * {@link org.apache.shiro.subject.Subject#isAuthenticated() subject.isAuthenticated()};
47       *
48       * @return true if the subject is authenticated; false if the subject is unauthenticated
49       */
50      protected boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) {
51          Subject subject = getSubject(request, response);
52          return subject.isAuthenticated();
53      }
54  
55      /**
56       * Redirects to user to the previously attempted URL after a successful login.  This implementation simply calls
57       * <code>{@link org.apache.shiro.web.util.WebUtils WebUtils}.{@link WebUtils#redirectToSavedRequest(javax.servlet.ServletRequest, javax.servlet.ServletResponse, String) redirectToSavedRequest}</code>
58       * using the {@link #getSuccessUrl() successUrl} as the {@code fallbackUrl} argument to that call.
59       *
60       * @param request  the incoming request
61       * @param response the outgoing response
62       * @throws Exception if there is a problem redirecting.
63       */
64  	protected void issueSuccessRedirect(ServletRequest request, ServletResponse response) throws Exception {
65  		String requestUri = getSuccessUrl();
66  		if (!requestUri.startsWith("/")) requestUri = "/" + requestUri;
67  		if (isRedirectToSavedUrl()) {
68  			getLoginContextService().redirectToSavedRequest(requestUri);
69  			return;
70  		}
71  		WebUtils.issueRedirect(request, response, requestUri);
72  	}
73  }
